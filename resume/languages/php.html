PHP Syntax + Data Structures

Quick Reference

Basic Variable Types

$x = 5;
$pi = 3.14;
$name = "Lukas";
$flag = true;

Output / Input

echo "Hello";
$name = readline("Enter name: ");

Conditionals

if ($x > 10) { }
elseif ($x == 10) { }
else { }

Loops

for ($i = 0; $i < 5; $i++) { }

while ($x > 0) { $x--; }

do { $x--; } while ($x > 0);

Functions

function add($a, $b) {
    return $a + $b;
}

Recursion

function factorial($n) {
    if ($n == 0) return 1;
    return $n * factorial($n - 1);
}

Array

$arr = [1, 2, 3];

Associative Array (Hash Map)

$map = ["a" => 1];
$map["b"] = 2;

Set (Using Array Keys)

$set = [];
$set[$value] = true;

Stack / Queue

// Stack
array_push($arr, 1);
array_pop($arr);

// Queue
array_push($arr, 1);
array_shift($arr);

Class (Object)

class Person {
    public $name;
}

Linked List Node

class ListNode {
    public $val;
    public $next;
}

Format Number as Money

$price = 1234.5;
echo "$" . number_format($price, 2);

PHP-Specific Features

Superglobals

$_GET["id"];
$_POST["name"];

String Concatenation

$full = $first . " " . $last;

Include Files

include "file.php";

Loose Typing

$x = "5" + 5; // 10

Start PHP Block

<?php
echo "Hello";
?>
