Quick Reference

Basic Variable Types

let name = "Lukas";

Console Output / Input

console.log("Hello");
// Browser input
let name = prompt("Enter name");

Conditionals

if (x > 10) { }
else if (x === 10) { }

For Loop

for (let i = 0; i < 5; i++) { }

While Loop

while (x > 0) { x--; }

Functions

function add(a, b) {
  return a + b;
}

Arrow Function

const add = (a, b) => a + b;

Recursion

function factorial(n) {
  if (n === 0) return 1;
  return n * factorial(n - 1);
}

Array

let arr = [1, 2, 3];
arr.push(4);
arr.pop();

Map (Hash Map)

let map = new Map();
map.set("a", 1);

Set

let set = new Set();

Stack / Queue

// Stack
let stack = [];
stack.push(1);
stack.pop();

// Queue
let queue = [];
queue.push(1);
queue.shift();

Class (Object)

class Person {
  constructor(name) {
    this.name = name;
  }
}

Linked List Node

class ListNode {
  constructor(val, next = null) {
    this.val = val;
    this.next = next;
  }
}

Binary Tree Node

class TreeNode {
  constructor(val, left = null, right = null) {
    this.val = val;
    this.left = left;
    this.right = right;
  }
}

Trie Node

class TrieNode {
  constructor() {
    this.children = {};
    this.isEnd = false;
  }
}

Format Number as Money

let price = 1234.5;
price.toLocaleString("en-US", { style: "currency", currency: "USD" });
