Functions

function add(a, b) result(res)
    integer, intent(in) :: a, b
    integer :: res
    res = a + b
end function

Recursion

recursive function fact(n) result(res)
    integer, intent(in) :: n
    integer :: res
    if (n == 0) then
        res = 1
    else
        res = n * fact(n-1)
    end if
end function
