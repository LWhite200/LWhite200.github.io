Python Syntax + Data Structures

Complete Quick Reference

Basic Variable Types

x = 5
y = 3.14
name = "Lukas"
flag = True

Console Output / Input

print("Hello")
name = input("Enter name: ")

Conditionals

if x > 10:
    print("Big")
elif x == 10:
    print("Equal")
else:
    print("Small")

For Loop

for i in range(5):
    print(i)

While Loop

while x > 0:
    x -= 1

Functions

def add(a, b):
    return a + b

result = add(2, 3)

Recursion

def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n - 1)

Array (Typed, Fixed)

import array
arr = array.array('i', [1, 2, 3])

List (Dynamic Array)

nums = [1, 2, 3]
nums.append(4)
nums.pop()

Stack

stack = []
stack.append(10)
stack.pop()

Queue

from collections import deque
q = deque()
q.append(1)
q.popleft()

Dictionary (Hash Map)

mp = {"a": 1}
mp["b"] = 2

Set

s = {1, 2}
s.add(3)

Class (Object)

class Person:
    def __init__(self, name):
        self.name = name

p = Person("Lukas")

Linked List Node

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

Binary Tree Node

class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

Trie Node

class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

Format Number as Money

price = 1234.5
formatted = f"${price:,.2f}"

Python-Specific Features

List Comprehension

squares = [x*x for x in range(5)]

Multiple Assignment

a, b = 5, 10
a, b = b, a

Enumerate

for i, val in enumerate(arr):
    print(i, val)

Lambda

add = lambda a, b: a + b

Slicing

arr[1:4]
arr[::-1]

Indentation Defines Scope

if x > 0:
    print("Inside block")
