C Syntax + Data Structures

Quick Reference

Basic Variable Types

int x = 5;
float pi = 3.14f;
char letter = 'A';
int flag = 1; // boolean style

Console Output / Input

printf("Hello\n");
scanf("%d", &x);

Conditionals

if (x > 10) { }
else if (x == 10) { }
else { }

For Loop

for (int i = 0; i < 5; i++) { }

While Loop

while (x > 0) { x--; }

Do-While Loop

do { x--; } while (x > 0);

Functions

int add(int a, int b) {
    return a + b;
}

Recursion

int factorial(int n) {
    if (n == 0) return 1;
    return n * factorial(n - 1);
}

Array

int arr[3] = {1, 2, 3};

Struct (Object-like)

struct Person {
    char name[50];
    int age;
};

Linked List Node

struct ListNode {
    int val;
    struct ListNode* next;
};

Binary Tree Node

struct TreeNode {
    int val;
    struct TreeNode* left;
    struct TreeNode* right;
};

Dynamic Memory

int* ptr = malloc(sizeof(int) * 5);
free(ptr);

Format Number as Money

double price = 1234.5;
printf("$%.2f", price);

    return 0;
